      statusCode: response.status,
      loadTime: timeTaken + " ms",
      contentLength: response.headers["content-length"] || "Unknown",
    });

  } catch (error) {
    res.json({
      status: "Offline",
      error: error.message,
    });
  }
});

import { useState } from "react";
import axios from "axios";

function App() {
  const [url, setUrl] = useState("");
  const [result, setResult] = useState(null);

  const checkWebsite = async () => {
    const res = await axios.post("http://localhost:5000/check", { url });
    setResult(res.data);
  };

  return (
    <div style={{ padding: "40px", fontFamily: "Arial" }}>
      <h1>Website Performance Monitoring Tool</h1>

      <input 
        type="text"
        placeholder="Enter website URL"
        value={url}
        onChange={(e) => setUrl(e.target.value)}
        style={{ padding: "10px", width: "300px" }}
      />

      <button 
        onClick={checkWebsite}
        style={{ padding: "10px 20px", marginLeft: "10px" }}
      >
        Check
      </button>

      {result && (
        <div style={{ marginTop: "30px" }}>
          <h2>Results:</h2>
          <p><b>Status:</b> {result.status}</p>
          <p><b>Status Code:</b> {result.statusCode}</p>
          <p><b>Load Time:</b> {result.loadTime}</p>
          <p><b>Content Length:</b> {result.contentLength}</p>
          {result.error && <p><b>Error:</b> {result.error}</p>}
        </div>
      )}
    </div>
  );
}

export default App;
